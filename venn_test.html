# Es posible realizar algebra de conjuntos con los objetos tipo conjunto,
# el operador & significa interseccion,
# el operador - significa diferencia,
# el operador | significa union:

universo = chocolate | vainilla | fresa
print 'Universo: {0}.\n'.format(', '.join(universo))

print 'Les gusta de los tres: {0}.\n'.format(', '.join(vainilla & fresa & chocolate))

print 'Les gusta de fresa y vainilla: {0}.'.format(', '.join(fresa & vainilla))
print 'Les gusta de fresa y vainilla pero no chocolate: {0}.\n'.format(', '.join((fresa & vainilla)-chocolate))

print 'Les gusta de vainilla y chocolate: {0}.'.format(', '.join(vainilla & chocolate))
print 'Les gusta de vainilla y chocolate pero no fresa: {0}.\n'.format(', '.join((vainilla & chocolate)-fresa))

print 'Les gusta de chocolate y fresa: {0}.'.format(', '.join(chocolate & vainilla))
# En este caso es 0 y un conjunto vacio, por eso no imprime nada:
print 'Les gusta de chocolate y fresa pero no vainilla: {0}.\n'.format(', '.join((chocolate & fresa)-vainilla))

print 'Les gusta unicamente de fresa: {0}.'.format(', '.join(fresa - (chocolate | vainilla)))
print 'Les gusta unicamente de chocolate: {0}.'.format(', '.join(chocolate - (fresa | vainilla)))
print 'Les gusta unicamente de vainilla: {0}.'.format(', '.join(vainilla - (chocolate | fresa)))
